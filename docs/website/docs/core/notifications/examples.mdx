import CodeBlock from '@theme/CodeBlock';

import {
  NotificationsProvider,
  useNotifications,
  TOAST_TYPES,
} from '@ttoss/notifications';

import { Button, ThemeProvider } from '@ttoss/ui';

export const theme = {
  buttons: {
    success: { color: 'white', backgroundColor: 'green' },
    warning: { color: 'white', backgroundColor: 'orange' },
  },
};

export const Wrapper = ({ children }) => (
  <ThemeProvider theme={theme}>
    <NotificationsProvider>{children}</NotificationsProvider>
  </ThemeProvider>
);

export const ToastButton = ({ type, children, variant }) => {
  const { toast } = useNotifications();
  return (
    <Wrapper>
      <div className="padding-vert--md">
        <Button
          sx={{ width: 200 }}
          variant={variant}
          onClick={() => toast('Toast ' + type, { type })}
        >
          {children}
        </Button>
      </div>
    </Wrapper>
  );
};
